api - onee.ajax

(1)ajax - 接口全字段

// @param file    {String}
// @param method  [String] 默认"GET"
// @param time    [Number] 默认启动超时
// @param cache   [Boolean] 默认开启
// @param data    [Object] 提交的数据
// @param done    [Function] 成功回调，会把数据传回
// @param err     [Function] 出错回调，会把状态码传回
// @param timeout [Function] 超时回调
onee.ajax("dd.json", {
	
	method: "POST",
	time  : 1e3,
	cache : !!0,
	type  : "json",
	data  : {name:"chen", job : "web developer"},
	done  : function (J) {
		console.log(J)
	},
	err   : function (code) {
		console.log(code)
	},
	timeout: function () {
		console.log("timeout")
	}
	
});

(2)getJSON - 接口全字段

// @param file    {String}
// @param done    [Function] 成功回调，会把数据传回
// @param err     [Function] 出错回调，会把状态码传回
// @param data    [Object] 提交的数据
onee.getJSON(
	"dd.json",
	function (J) {
		console.log(J)
	},
	function (ecode) {
		console.log(ecode)
	},
	{name:"chen", job : "web developer"}
);

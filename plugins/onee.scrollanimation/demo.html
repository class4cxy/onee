onee.use("onee.scrollanimation").done(function () {
    onee.scrollanimation("#animation1", {
       // 启动鼠标滚轮切换，默认false
        enableWheel : !!1,
       // 启动触点导航控件，默认true
        enableNavigator : !!1,
       // 初始化时显示的场景，默认第一帧场景，由0开始
        currentScene : 0,
       // 启动自动播放，默认不启动
        autoPlay : !!1,
       // 场景间切换时长
        playDuration : 1500,
       // 场景切换方向[x|y]，默认延x轴
        scrollDiration : "y"
       // 自动播放间隔时长
        autoPlayDuration : 5e3,
    });
})

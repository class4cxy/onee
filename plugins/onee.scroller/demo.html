        playDuration : 1500,

        scrollDiration : "y"
       // 自动播放间隔时长
        autoPlayDuration : 5e3,
       // 场景节点活动时回调
        onSceneActive : [
           function () {},
           function () {},
           function () {},
           function () {},
           function () {}
       ]
    });
})
